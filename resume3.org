  (defun traverse (tree predicate action)
    (cond ((null tree)  nil)
          ((listp tree) (if (funcall predicate tree)
                            (funcall action tree)))
          (t (message "= error2: %s" tree))))

  (defun process-headline (param)
    (let ((name (getf (car param) :raw-value)))
      (cond ((equal name "Info")
             (process-info (cdr param)))
            ((equal name "Educations")
             (process-educations (cdr param)))
            ((equal name "Expiriences")
             (process-expiriences (cdr param)))
            ((equal name "Languages")
             (process-languages (cdr param)))
            ((equal name "Contacts")
             (process-contacts (cdr param)))
            ((equal name "About")
             (process-about (cdr param)))
      (message ">> unknown header: %s - %s"
               (getf (car param) :raw-value)
               (length (cdr param))))))

  (defun process-info (param)
    (message "-> info"))

  (defun process-educations (param)
    (message "-> educations"))

  (defun process-expiriences (param)
    (message "-> expiriences"))

  (defun process-languages (param)
    (message "-> languages"))

  (defun process-contacts (param)
    (message "-> contacts"))

  (defun process-about (param)
    (message "--> about %s" (car (car param)))
    (process-about-section (cddar param)))

  (let ((filename "~/repo/moto/resume2.org"))

                  #'(lambda (x)
                      (let ((lead (car tree)))
                        (cond ((equal lead 'headline)
                               (process-headline (cdr tree)))
                              ((listp lead)
                               (let ((fst (traverse (car tree) predicate action)))
                                 (if (null fst)
                                     (traverse (cdr tree) predicate action)
                                   fst)))
                              (t (traverse (cddr tree) predicate action)))))
                  #'(lambda (x)))
        tree))
    nil)

  (defun process-about-section (param)
    ;; (message "---> about-section-contents %s" param)
    (mapcar #'(lambda (par)
                (let ((lead (car par)))
                  (cond ((equal lead 'paragraph)
                         (message "----> [%s]" lead)
                         )
                        ((equal lead 'plain-list)
                         (message "----> [%s]" lead)
                         )
                        (t (message "----> [%s]" lead)))
                  ))
            param)
    )
